
--Creates table users
create table users(uid serial primary key not null, utype varchar(20) not null, user_name varchar(50) not null, user_Lname varchar(50) not null, Address1 varchar(100) not null, city varchar(50) not null, State varchar(25) not null, zipcode varchar(10) not null, country varchar(20) not null, phone varchar(20) not null, user_email varchar(30) UNIQUE not null , user_nickname varchar(25) unique not null, user_password text not null, user_created timestamp not null);

--Creates table Category
create table Category(cid serial primary key not null, cat_name varchar(100) not null);


-- Creates table Products
create table Products(pid serial primary key not null, pname varchar(50) not null, description varchar(500), pprice numeric(7, 2) CHECK (pprice >0) not null, shipping_cost numeric(7,2) Check(shipping_cost >0)not null,cid integer references Category(cid) not null, uid integer references users(uid));

--Create table Cart
create table Cart(caid serial primary key not null, cart_id integer, uid integer references Users(uid), pid integer references Products(pid) not null, p_qty integer DEFAULT 0);

--Creates table Orders
create table Orders(order_id serial primary key not null, cart_id integer references Cart(caid), uid integer references Users(uid));

--Create relation Posts
create table Posts(uid integer references users(uid), pid integer references Products(pid), qty integer, primary key(uid,pid));


--Create relation Ships_to
create table Ships_to(uid integer references users(uid), order_id integer references Orders(order_id), primary key(uid,order_id));

--Create table Events
create table Events(event_id serial primary key, event varchar(500), e_description varchar(1000), e_date date, e_location varchar(200));
